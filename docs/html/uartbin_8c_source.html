133static uartbin_status_t uartbin_build_frame(uint8_t *frame,
134 size_t frame_capacity,
135 uint8_t type,
136 uint8_t flags,
137 uint16_t seq,
138 const uint8_t *payload,
139 uint16_t payload_len,
140 size_t *frame_len)
141{
142 size_t len = uartbin_frame_len(payload_len);
143 uint16_t crc;
144
145 if (frame == 0 || frame_len == 0 || (payload_len > 0u && payload == 0)) {
146 return UARTBIN_EINVAL;
147 }
148 if (len > frame_capacity) {
149 return UARTBIN_EPAYLOAD_TOO_LONG;
150 }
151
152 frame[0] = UARTBIN_SOF0;
153 frame[1] = UARTBIN_SOF1;
154 frame[2] = UARTBIN_VERSION;
155 frame[3] = type;
156 frame[4] = flags;
157 frame[5] = 0u;
158 uartbin_put_u16_le(&frame[6], seq);
159 uartbin_put_u16_le(&frame[8], payload_len);
160 if (payload_len > 0u) {
161 memcpy(&frame[10], payload, payload_len);
162 }
163
164 crc = uartbin_crc16_ccitt(&frame[2], UARTBIN_HEADER_SIZE, 0xFFFFu);
165 crc = uartbin_crc16_ccitt(payload, payload_len, crc);
166 uartbin_put_u16_le(&frame[10u + payload_len], crc);
167
168 *frame_len = len;
169 return UARTBIN_OK;
170}

244static void uartbin_poll_retry(uartbin_t *ctx, uint32_t now_ms)
245{
246 uint32_t elapsed;
247
248 if (ctx->retry_active == 0u || !uartbin_retry_enabled(ctx)) {
249 return;
250 }
251
252 if (ctx->retry_timer_started == 0u) {
253 ctx->retry_timer_started = 1u;
254 ctx->retry_last_tx_time_ms = now_ms;
255 return;
256 }
257
258 elapsed = now_ms - ctx->retry_last_tx_time_ms;
259 if (elapsed < ctx->cfg.tx_retry_timeout_ms) {
260 return;
261 }
262
263 if (ctx->retry_count >= ctx->cfg.tx_retry_max_retries) {
264 uartbin_report_error(ctx, UARTBIN_ERROR_RETRY_EXHAUSTED);
265 uartbin_clear_retry(ctx);
266 return;
267 }
268
269 if (uartbin_write_all(ctx, ctx->cfg.tx_retry_buffer, ctx->retry_frame_len) != UARTBIN_OK) {
270 uartbin_report_error(ctx, UARTBIN_ERROR_RETRY_WRITE);
271 uartbin_clear_retry(ctx);
272 return;
273 }
274
275 ctx->retry_count++;
276 ctx->retry_last_tx_time_ms = now_ms;
277}

320uint16_t uartbin_crc16_ccitt(const uint8_t *data, size_t len, uint16_t seed)
321{
322 uint16_t crc = seed;
323
324 while (len > 0u) {
325 crc ^= (uint16_t)(*data++) << 8;
326 len--;
327 for (uint8_t bit = 0u; bit < 8u; bit++) {
328 if ((crc & 0x8000u) != 0u) {
329 crc = (uint16_t)((crc << 1) ^ 0x1021u);
330 } else {
331 crc <<= 1;
332 }
333 }
334 }
335
336 return crc;
337}

339uartbin_status_t uartbin_send(uartbin_t *ctx,
340 uint8_t type,
341 uint8_t flags,
342 uint16_t seq,
343 const uint8_t *payload,
344 uint16_t payload_len)
345{
346 uint8_t sof[2] = { UARTBIN_SOF0, UARTBIN_SOF1 };
347 uint8_t header[UARTBIN_HEADER_SIZE];
348 uint8_t crc_bytes[UARTBIN_CRC_SIZE];
349 uint16_t crc;
350 uartbin_status_t status;
351
352 if (ctx == 0 || (payload_len > 0u && payload == 0)) {
353 return UARTBIN_EINVAL;
354 }
355
356 header[0] = UARTBIN_VERSION;
357 header[1] = type;
358 header[2] = flags;
359 header[3] = 0u;
360 uartbin_put_u16_le(&header[4], seq);
361 uartbin_put_u16_le(&header[6], payload_len);
362
363 crc = uartbin_crc16_ccitt(header, sizeof(header), 0xFFFFu);
364 crc = uartbin_crc16_ccitt(payload, payload_len, crc);
365 uartbin_put_u16_le(crc_bytes, crc);
366
367 status = uartbin_write_all(ctx, sof, sizeof(sof));
368 if (status != UARTBIN_OK) {
369 return status;
370 }
371
372 status = uartbin_write_all(ctx, header, sizeof(header));
373 if (status != UARTBIN_OK) {
374 return status;
375 }
376
377 status = uartbin_write_all(ctx, payload, payload_len);
378 if (status != UARTBIN_OK) {
379 return status;
380 }
381
382 return uartbin_write_all(ctx, crc_bytes, sizeof(crc_bytes));
383}

533void uartbin_feed_byte_at(uartbin_t *ctx, uint8_t byte, uint32_t now_ms)
534{
535 if (ctx == 0) {
536 return;
537 }
538
539 uartbin_poll_rx_timeout(ctx, now_ms);
540 ctx->last_rx_time_ms = now_ms;
541
542 switch (ctx->state) {
543 case UARTBIN_STATE_SOF0:
544 /* Ilk start-of-frame byte'i gorunene kadar noise ignore edilir. */
545 if (byte == UARTBIN_SOF0) {
546 ctx->state = UARTBIN_STATE_SOF1;
547 }
548 break;
549
550 case UARTBIN_STATE_SOF1:
551 /*
552 * Tekrarlanan SOF0 bizi SOF1 icin hazir tutar. Bu, A5 A5 5A gibi byte
553 * akislarinda ikinci A5'i atmadan yeniden senkron olmaya yardim eder.
554 */
555 if (byte == UARTBIN_SOF1) {
556 ctx->state = UARTBIN_STATE_HEADER;
557 ctx->header_pos = 0u;
558 ctx->crc = 0xFFFFu;
559 } else if (byte != UARTBIN_SOF0) {
560 ctx->state = UARTBIN_STATE_SOF0;
561 }
562 break;
563
564 case UARTBIN_STATE_HEADER:
565 /* Header sabit boyutludur; uzunluk ve surum erken dogrulanabilir. */
566 ctx->header[ctx->header_pos++] = byte;
567 if (ctx->header_pos == UARTBIN_HEADER_SIZE) {
568 ctx->crc = uartbin_crc16_ccitt(ctx->header, UARTBIN_HEADER_SIZE, ctx->crc);
569 ctx->payload_len = uartbin_get_u16_le(&ctx->header[6]);
570 ctx->payload_pos = 0u;
571
572 if (ctx->header[0] != UARTBIN_VERSION) {
573 uartbin_report_error(ctx, UARTBIN_ERROR_BAD_VERSION);
574 uartbin_reset_rx(ctx);
575 } else if (ctx->payload_len > ctx->cfg.rx_payload_capacity) {
576 uartbin_report_error(ctx, UARTBIN_ERROR_BAD_LENGTH);
577 uartbin_reset_rx(ctx);
578 } else if (ctx->payload_len > 0u && ctx->cfg.rx_payload_buffer == 0) {
579 uartbin_report_error(ctx, UARTBIN_ERROR_RX_OVERFLOW);
580 uartbin_reset_rx(ctx);
581 } else {
582 ctx->state = (ctx->payload_len == 0u) ? UARTBIN_STATE_CRC : UARTBIN_STATE_PAYLOAD;
583 ctx->crc_pos = 0u;
584 }
585 }
586 break;
587
588 case UARTBIN_STATE_PAYLOAD:
589 /* Bu duruma girmeden uzunluk dogrulandigi icin yazma guvenlidir. */
590 ctx->cfg.rx_payload_buffer[ctx->payload_pos] = byte;
591 ctx->payload_pos++;
592 ctx->crc = uartbin_crc16_ccitt(&byte, 1u, ctx->crc);
593 if (ctx->payload_pos == ctx->payload_len) {
594 ctx->state = UARTBIN_STATE_CRC;
595 ctx->crc_pos = 0u;
596 }
597 break;
598
599 case UARTBIN_STATE_CRC:
600 /* Packet callback yalnizca alinan CRC eslestikten sonra cagrilir. */
601 ctx->crc_bytes[ctx->crc_pos++] = byte;
602 if (ctx->crc_pos == UARTBIN_CRC_SIZE) {
603 uint16_t expected = uartbin_get_u16_le(ctx->crc_bytes);
604
605 if (expected == ctx->crc) {
606 uartbin_packet_t packet;
607
608 packet.type = ctx->header[1];
609 packet.flags = ctx->header[2];
610 packet.seq = uartbin_get_u16_le(&ctx->header[4]);
611 packet.payload = ctx->cfg.rx_payload_buffer;
612 packet.payload_len = ctx->payload_len;
613 if ((packet.flags & UARTBIN_FLAG_RESPONSE) != 0u &&
614 ctx->retry_active != 0u &&
615 packet.seq == ctx->retry_seq) {
616 uartbin_clear_retry(ctx);
617 }
618
619 if (ctx->cfg.on_packet != 0) {
620 ctx->cfg.on_packet(&packet, ctx->cfg.user);
621 }
622 } else {
623 uartbin_report_error(ctx, UARTBIN_ERROR_CRC);
624 }
625
626 uartbin_reset_rx(ctx);
627 }
628 break;
629
630 default:
631 uartbin_reset_rx(ctx);
632 break;
633 }

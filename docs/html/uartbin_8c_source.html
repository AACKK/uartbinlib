146uint16_t uartbin_crc16_ccitt(const uint8_t *data, size_t len, uint16_t seed)
147{
148 uint16_t crc = seed;
149
150 while (len-- > 0u) {
151 crc ^= (uint16_t)(*data++) << 8;
152 for (uint8_t bit = 0u; bit < 8u; bit++) {
153 if ((crc & 0x8000u) != 0u) {
154 crc = (uint16_t)((crc << 1) ^ 0x1021u);
155 } else {
156 crc <<= 1;
157 }
158 }
159 }
160
161 return crc;
162}

164uartbin_status_t uartbin_send(uartbin_t *ctx,
165 uint8_t type,
166 uint8_t flags,
167 uint16_t seq,
168 const uint8_t *payload,
169 uint16_t payload_len)
170{
171 uint8_t sof[2] = { UARTBIN_SOF0, UARTBIN_SOF1 };
172 uint8_t header[UARTBIN_HEADER_SIZE];
173 uint8_t crc_bytes[UARTBIN_CRC_SIZE];
174 uint16_t crc;
175 uartbin_status_t status;
176
177 if (ctx == 0 || (payload_len > 0u && payload == 0)) {
178 return UARTBIN_EINVAL;
179 }
180
181 header[0] = UARTBIN_VERSION;
182 header[1] = type;
183 header[2] = flags;
184 header[3] = 0u;
185 uartbin_put_u16_le(&header[4], seq);
186 uartbin_put_u16_le(&header[6], payload_len);
187
188 crc = uartbin_crc16_ccitt(header, sizeof(header), 0xFFFFu);
189 crc = uartbin_crc16_ccitt(payload, payload_len, crc);
190 uartbin_put_u16_le(crc_bytes, crc);
191
192 status = uartbin_write_all(ctx, sof, sizeof(sof));
193 if (status != UARTBIN_OK) {
194 return status;
195 }
196
197 status = uartbin_write_all(ctx, header, sizeof(header));
198 if (status != UARTBIN_OK) {
199 return status;
200 }
201
202 status = uartbin_write_all(ctx, payload, payload_len);
203 if (status != UARTBIN_OK) {
204 return status;
205 }
206
207 return uartbin_write_all(ctx, crc_bytes, sizeof(crc_bytes));
208}

237void uartbin_poll(uartbin_t *ctx, uint32_t now_ms)
238{
239 uint32_t elapsed;
240
241 if (ctx == 0 || ctx->cfg.rx_timeout_ms == 0u || !uartbin_rx_active(ctx)) {
242 return;
243 }
244
245 elapsed = now_ms - ctx->last_rx_time_ms;
246 if (elapsed >= ctx->cfg.rx_timeout_ms) {
247 uartbin_report_error(ctx, UARTBIN_ERROR_TIMEOUT);
248 uartbin_reset_rx(ctx);
249 ctx->last_rx_time_ms = now_ms;
250 }

259void uartbin_feed_byte_at(uartbin_t *ctx, uint8_t byte, uint32_t now_ms)
260{
261 if (ctx == 0) {
262 return;
263 }
264
265 uartbin_poll(ctx, now_ms);
266 ctx->last_rx_time_ms = now_ms;
267
268 switch (ctx->state) {
269 case UARTBIN_STATE_SOF0:
270 /* Ignore noise until the first start-of-frame byte appears. */
271 if (byte == UARTBIN_SOF0) {
272 ctx->state = UARTBIN_STATE_SOF1;
273 }
274 break;
275
276 case UARTBIN_STATE_SOF1:
277 /*
278 * A repeated SOF0 keeps us ready for SOF1. This helps resync on byte
279 * streams such as A5 A5 5A without throwing away the second A5.
280 */
281 if (byte == UARTBIN_SOF1) {
282 ctx->state = UARTBIN_STATE_HEADER;
283 ctx->header_pos = 0u;
284 ctx->crc = 0xFFFFu;
285 } else if (byte != UARTBIN_SOF0) {
286 ctx->state = UARTBIN_STATE_SOF0;
287 }
288 break;
289
290 case UARTBIN_STATE_HEADER:
291 /* Header is fixed-size, so length and version can be validated early. */
292 ctx->header[ctx->header_pos++] = byte;
293 if (ctx->header_pos == UARTBIN_HEADER_SIZE) {
294 ctx->crc = uartbin_crc16_ccitt(ctx->header, UARTBIN_HEADER_SIZE, ctx->crc);
295 ctx->payload_len = uartbin_get_u16_le(&ctx->header[6]);
296 ctx->payload_pos = 0u;
297
298 if (ctx->header[0] != UARTBIN_VERSION) {
299 uartbin_report_error(ctx, UARTBIN_ERROR_BAD_VERSION);
300 uartbin_reset_rx(ctx);
301 } else if (ctx->payload_len > ctx->cfg.rx_payload_capacity) {
302 uartbin_report_error(ctx, UARTBIN_ERROR_BAD_LENGTH);
303 uartbin_reset_rx(ctx);
304 } else if (ctx->payload_len > 0u && ctx->cfg.rx_payload_buffer == 0) {
305 uartbin_report_error(ctx, UARTBIN_ERROR_RX_OVERFLOW);
306 uartbin_reset_rx(ctx);
307 } else {
308 ctx->state = (ctx->payload_len == 0u) ? UARTBIN_STATE_CRC : UARTBIN_STATE_PAYLOAD;
309 ctx->crc_pos = 0u;
310 }
311 }
312 break;
313
314 case UARTBIN_STATE_PAYLOAD:
315 /* Length was validated before entering this state, so the write is safe. */
316 ctx->cfg.rx_payload_buffer[ctx->payload_pos] = byte;
317 ctx->payload_pos++;
318 ctx->crc = uartbin_crc16_ccitt(&byte, 1u, ctx->crc);
319 if (ctx->payload_pos == ctx->payload_len) {
320 ctx->state = UARTBIN_STATE_CRC;
321 ctx->crc_pos = 0u;
322 }
323 break;
324
325 case UARTBIN_STATE_CRC:
326 /* The packet callback is called only after the received CRC matches. */
327 ctx->crc_bytes[ctx->crc_pos++] = byte;
328 if (ctx->crc_pos == UARTBIN_CRC_SIZE) {
329 uint16_t expected = uartbin_get_u16_le(ctx->crc_bytes);
330
331 if (expected == ctx->crc) {
332 if (ctx->cfg.on_packet != 0) {
333 uartbin_packet_t packet;
334 packet.type = ctx->header[1];
335 packet.flags = ctx->header[2];
336 packet.seq = uartbin_get_u16_le(&ctx->header[4]);
337 packet.payload = ctx->cfg.rx_payload_buffer;
338 packet.payload_len = ctx->payload_len;
339 ctx->cfg.on_packet(&packet, ctx->cfg.user);
340 }
341 } else {
342 uartbin_report_error(ctx, UARTBIN_ERROR_CRC);
343 }
344
345 uartbin_reset_rx(ctx);
346 }
347 break;
348
349 default:
350 uartbin_reset_rx(ctx);
351 break;
352 }

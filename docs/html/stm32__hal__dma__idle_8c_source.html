97static int stm32_uartbin_dma_start_rx(stm32_uartbin_dma_t *port)
98{
99 int status = HAL_UARTEx_ReceiveToIdle_DMA(port->huart,
100 port->dma_rx_buffer,
101 sizeof(port->dma_rx_buffer));
102 if (status == HAL_OK) {
103 __HAL_DMA_DISABLE_IT(port->hdma_rx, DMA_IT_HT);
104 return 0;
105 }
106
107 return -1;
108}

161void stm32_uartbin_dma_init(stm32_uartbin_dma_t *port,

194void stm32_uartbin_dma_rx_event_callback(stm32_uartbin_dma_t *port,
195 UART_HandleTypeDef *huart,
196 uint16_t size)
197{
198 if (huart == port->huart) {
199 if (size <= sizeof(port->dma_rx_buffer)) {
200 uartbin_feed_at(&port->link, port->dma_rx_buffer, size, HAL_GetTick());
201 } else {
202 port->hal_rx_errors++;
203 uartbin_reset(&port->link);
204 }
205
206 if (stm32_uartbin_dma_start_rx(port) != 0) {
207 port->hal_rx_errors++;
208 uartbin_reset(&port->link);
209 }
210 }
211}

100static int stm32_uartbin_dma_start_rx(stm32_uartbin_dma_t *port)

164void stm32_uartbin_dma_init(stm32_uartbin_dma_t *port,
165 UART_HandleTypeDef *huart,

168 uartbin_config_t cfg;

172 port->hal_rx_errors = 0u;

174
175 memset(&cfg, 0, sizeof(cfg));
176 cfg.write = stm32_uartbin_dma_write;
177 cfg.on_packet = stm32_uartbin_dma_on_packet;
178 cfg.on_error = stm32_uartbin_dma_on_error;
179 cfg.user = port;
180 cfg.rx_payload_buffer = port->rx_payload;
181 cfg.rx_payload_capacity = sizeof(port->rx_payload);
182 cfg.rx_timeout_ms = 50u;
183
184 uartbin_init(&port->link, &cfg);
185 (void)stm32_uartbin_dma_start_rx(port);
186}

198void stm32_uartbin_dma_rx_event_callback(stm32_uartbin_dma_t *port,
199 UART_HandleTypeDef *huart,
200 uint16_t size)
201{
202 if (huart == port->huart) {
203 if (size <= sizeof(port->dma_rx_buffer)) {
204 uartbin_feed_at(&port->link, port->dma_rx_buffer, size, HAL_GetTick());
205 } else {
206 port->hal_rx_errors++;
207 uartbin_reset(&port->link);
208 }
209
210 if (stm32_uartbin_dma_start_rx(port) != 0) {
211 port->hal_rx_errors++;

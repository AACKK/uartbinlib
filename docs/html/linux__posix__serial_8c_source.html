99static int linux_serial_configure(int fd, unsigned baud)
100{
101 struct termios tio;
102 speed_t speed = linux_baud_to_speed(baud);
103
104 if (tcgetattr(fd, &tio) != 0) {
105 return -1;
106 }
107
108 cfmakeraw(&tio);
109 tio.c_cflag |= (CLOCAL | CREAD);
110 tio.c_cflag &= (tcflag_t)~CSTOPB;
111 tio.c_cflag &= (tcflag_t)~PARENB;
112 tio.c_cflag &= (tcflag_t)~CSIZE;
113 tio.c_cflag |= CS8;
114 tio.c_cc[VMIN] = 0;
115 tio.c_cc[VTIME] = 0;
116
117 if (cfsetispeed(&tio, speed) != 0 || cfsetospeed(&tio, speed) != 0) {
118 return -1;
119 }
120 if (tcsetattr(fd, TCSANOW, &tio) != 0) {
121 return -1;
122 }
123
124 return 0;
125}

133static int linux_uartbin_write(const uint8_t *data, size_t len, void *user)
134{
135 linux_uartbin_t *port = (linux_uartbin_t *)user;
136 size_t sent = 0u;
137
138 while (sent < len) {
139 ssize_t n = write(port->fd, &data[sent], len - sent);
140 if (n > 0) {
141 sent += (size_t)n;
142 continue;
143 }
144 if (n < 0 && (errno == EINTR)) {
145 continue;
146 }
147 if (n < 0 && (errno == EAGAIN || errno == EWOULDBLOCK)) {
148 struct pollfd pfd;
149
150 pfd.fd = port->fd;
151 pfd.events = POLLOUT;
152 pfd.revents = 0;
153 if (poll(&pfd, 1u, 100) > 0 && (pfd.revents & POLLOUT) != 0) {
154 continue;
155 }
156 }
157
158 return -1;
159 }
160
161 return 0;
162}

198static int linux_uartbin_init(linux_uartbin_t *port, const char *path, unsigned baud)
199{
200 uartbin_config_t cfg;
201
202 memset(port, 0, sizeof(*port));
203 port->fd = open(path, O_RDWR | O_NOCTTY | O_NONBLOCK);
204 if (port->fd < 0) {
205 return -1;
206 }
207 if (linux_serial_configure(port->fd, baud) != 0) {
208 close(port->fd);
209 port->fd = -1;
210 return -1;
211 }
212
213 memset(&cfg, 0, sizeof(cfg));
214 cfg.write = linux_uartbin_write;
215 cfg.on_packet = linux_uartbin_on_packet;
216 cfg.on_error = linux_uartbin_on_error;
217 cfg.user = port;
218 cfg.rx_payload_buffer = port->rx_payload;
219 cfg.rx_payload_capacity = sizeof(port->rx_payload);
220 cfg.rx_timeout_ms = 50u;
221 cfg.tx_retry_buffer = port->tx_retry_frame;
222 cfg.tx_retry_capacity = sizeof(port->tx_retry_frame);
223 cfg.tx_retry_timeout_ms = UARTBIN_DEFAULT_RETRY_TIMEOUT_MS;
224 cfg.tx_retry_max_retries = UARTBIN_DEFAULT_RETRY_MAX_RETRIES;
225
226 uartbin_init(&port->link, &cfg);
227 return 0;
228}

233static int linux_uartbin_run(linux_uartbin_t *port)
234{
235 for (;;) {
236 struct pollfd pfd;
237 int rc;
238
239 pfd.fd = port->fd;
240 pfd.events = POLLIN | POLLERR | POLLHUP;
241 pfd.revents = 0;
242
243 rc = poll(&pfd, 1u, 10);
244 if (rc < 0 && errno == EINTR) {
245 continue;
246 }
247 if (rc < 0) {
248 return -1;
249 }
250
251 if (rc > 0 && (pfd.revents & POLLIN) != 0) {
252 ssize_t n = read(port->fd, port->rx_block, sizeof(port->rx_block));
253 if (n > 0) {
254 uartbin_feed_at(&port->link, port->rx_block, (size_t)n, linux_millis());
255 } else if (n < 0 && errno != EAGAIN && errno != EWOULDBLOCK && errno != EINTR) {
256 uartbin_reset(&port->link);
257 }
258 }
259
260 if (rc > 0 && (pfd.revents & (POLLERR | POLLHUP)) != 0) {
261 uartbin_reset(&port->link);
262 }
263
264 uartbin_poll(&port->link, linux_millis());
265 }
266}
